    int  *ball = new int[n+1] ;

    for ( i = 0 ; i < n+1 ; ++i ) ball[i] = 0 ;

    // 代刚杭u]`计q
    const int M = 100000 ;

    // 代刚 M Ω
    for ( i = 0 ; i < M ; ++i ) {

        // p衡u]bgL n hu近岐害旄m
        position = 0 ;
        for ( j = 0 ; j < n ; ++j ) {
            // Y睹计 1 hy疾旌u近岍攻k, _h┕オ
            if ( rand()%2 ) ++position ;

        ++ball[position] ;

    int s = 1 ;
    for ( i = 0 ; i < n ; ++i ) s*=2 ;

        cout << "| " << static_cast<char>('A'+i) << " " ;
    cout << "|" << "\n\n" ;

    for ( i = 0 ; i < n + 1 ; ++i )
        cout << setw(3)
             << static_cast<unsigned>(1.*ball[i]*s/M+0.5) << " " ;
    cout << endl ;

    for ( i = 0 ; i < n + 1 ; ++i )  cout << setw(3) << " -- " ;
    cout << endl ;

    for ( i = 0 ; i < n + 1 ; ++i )  cout << setw(3) << s << " " ;
    cout << endl ;


    delete [] ball ;

    return 0 ;

}
