#include <iostream>
#include <algorithm>
#include <iterator>
#include <vector>

using namespace std ;

// 计CㄧΑン摸O
struct  Sequence {

    // 计C_l然P患W
    int  no , step ;  

    Sequence( int a = 0 , int s = 1 ) : no(a) , step(s) {}

    int  operator() ()  {
        int s = no ;
        no += step ;
        return s ;
    }

};


int main() {

    // a Τ 5 鹰缉w俱计
    vector<int>  a(5) ;

    // N a 铅C]wΘ : 1 2 3 4 5 
    generate(a.begin(),a.end(),Sequence(1,1)) ;

    // CL a 铅C
    copy(a.begin(),a.end(),ostream_iterator<int>(cout," ")) ;
    cout << endl ;

    // N a 铅CソЮ[W 4 3 2 1 单|蛹
    generate_n(back_inserter(a),4,Sequence(4,-1)) ;

    // CL a 铅C
    copy(a.begin(),a.end(),ostream_iterator<int>(cout," ")) ;
    cout << endl ;

    return 0 ;
}
